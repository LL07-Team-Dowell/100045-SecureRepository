      const repositoryCommits = {};

      apiResponse.data.forEach((item) => {
        const { pusher, backup_time, repository_name } = item;
        const month = new Date(backup_time).toLocaleString('default', { month: 'short' });
        const key = `${repository_name}-${month}`;

        if (repositoryCommits[key]) {
          repositoryCommits[key]++;
        } else {
          repositoryCommits[key] = 1;
        }
      });

      const newData = Object.keys(repositoryCommits).map((key) => {
        const [repository_name, month] = key.split('-');
        return {
          name: repository_name,
          month,
          commits: repositoryCommits[key],
        };
      });

      setRepositoryData(newData);
    }
  }, []);
